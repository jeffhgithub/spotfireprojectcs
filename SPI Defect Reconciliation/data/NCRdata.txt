SELECT 
    CONTAINER.CONTAINERNAME ModuleSN, 
    PRODUCTBASE_1.PRODUCTNAME "SAP Part Number", 
    PRODUCT_1.ZBNAMEOLD Product, 
    MFGORDER.MFGORDERNAME MfgOrder, 
    NONCONFORMANCEREPORT.NONCONFORMANCEREPORTNAME NCR_Report, 
    NONCONFORMANCEREPORT.CREATIONDATE NCR_CreationDate, 
    NONCONFORMANCEREPORT.NCRSTATUS NCR_Status, 
    NONCONFORMANCEREPORT.NCRCOMMENT NCR_Comment, 
    NONCONFORMANCEREPORT.ZCNCREFERENCEDESIGNATOR NCR_RefDes, 
    NONCONFORMANCEREPORT.ZCFMREFERENCEDESIGNATOR FailureMode_RefDes, 
    CONTAINER.CONTAINERID, 
    PRODUCT_1.PRODUCTID, 
    MFGORDER.MFGORDERID, 
    PRODUCTBASE_1.PRODUCTBASEID
FROM      
    /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCTBASE PRODUCTBASE_1 INNER JOIN 
        /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCT PRODUCT_1 INNER JOIN 
            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/MFGORDER MFGORDER FULL OUTER JOIN 
                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/CONTAINER CONTAINER INNER JOIN 
                    /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCT PRODUCT RIGHT OUTER JOIN 
                        /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCROOTCAUSECODE ZCROOTCAUSECODE RIGHT OUTER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCNONCONFORMANCECODE ZCNONCONFORMANCECODE RIGHT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/NONCONFORMANCEREPORT NONCONFORMANCEREPORT
                            ON ZCNONCONFORMANCECODE.ZCNONCONFORMANCECODEID = NONCONFORMANCEREPORT.ZCNONCONFORMANCECODEID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCFAILUREMODECODE ZCFAILUREMODECODE
                            ON NONCONFORMANCEREPORT.ZCFAILUREMODECODEID = ZCFAILUREMODECODE.ZCFAILUREMODECODEID
                        ON ZCROOTCAUSECODE.ZCROOTCAUSECODEID = NONCONFORMANCEREPORT.ZCROOTCAUSECODEID LEFT OUTER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCROOTCAUSECATEGORYCODE ZCROOTCAUSECATEGORYCODE
                        ON NONCONFORMANCEREPORT.ZCROOTCAUSECATEGORYCODEID = ZCROOTCAUSECATEGORYCODE.ZCROOTCAUSECATEGORYCODEID LEFT OUTER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCDISPOSITIONCODE ZCDISPOSITIONCODE
                        ON NONCONFORMANCEREPORT.ZCDISPOSITIONCODEID = ZCDISPOSITIONCODE.ZCDISPOSITIONCODEID LEFT OUTER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/WORKFLOWSTEP WORKFLOWSTEP
                        ON NONCONFORMANCEREPORT.CHARGETOSTEPID = WORKFLOWSTEP.WORKFLOWSTEPID FULL OUTER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZCROOTCAUSETYPE ZCROOTCAUSETYPE
                        ON ZCROOTCAUSECODE.ZCROOTCAUSETYPEID = ZCROOTCAUSETYPE.ZCROOTCAUSETYPEID
                    ON PRODUCT.PRODUCTID = NONCONFORMANCEREPORT.ZCFAILUREMODEPRODUCTID LEFT OUTER JOIN 
                        /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCTBASE PRODUCTBASE
                    ON PRODUCT.PRODUCTBASEID = PRODUCTBASE.PRODUCTBASEID
                ON CONTAINER.CONTAINERID = NONCONFORMANCEREPORT.CONTAINERID
            ON MFGORDER.MFGORDERID = CONTAINER.MFGORDERID
        ON PRODUCT_1.PRODUCTID = MFGORDER.PRODUCTID
    ON PRODUCT_1.PRODUCTBASEID = PRODUCTBASE_1.PRODUCTBASEID
