SELECT DISTINCT 
    CONTAINERLEVEL.CONTAINERLEVELNAME ContainerLevelName, 
    PRODUCT.DESCRIPTION Description, 
    MFGORDER.MFGORDERNAME MfgOrderName, 
    OPERATION.OPERATIONNAME OperationName, 
    CONTAINER.CONTAINERNAME PanelContainerName, 
    PRODUCTBASE.PRODUCTNAME ProductName, 
    RESOURCEDEF.RESOURCENAME ResourceName, 
    HISTORYMAINLINE.TXNDATE, 
    TestResultJudgement.ZBJUDGEMENTTYPENAME, 
    PRODUCT.ZBNAMEOLD, 
    ZBTESTPLANHISTORY.ZBTESTPLANNAME, 
    ZBTESTPLANHISTORY.ZBTESTPLANREVISION, 
    ZBTESTRESULTHISTORY.ZBTESTRESULTNAME, 
    ZBTESTRESULTHISTORY.ZBTESTRESULTROWINDEX, 
    ZBTESTRESULTHISTORY.ZBTESTRESULTVALUE
FROM      
    /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/RESOURCEDEF RESOURCEDEF INNER JOIN 
        /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/CONTAINERLEVEL CONTAINERLEVEL INNER JOIN 
            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCTBASE PRODUCTBASE INNER JOIN 
                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/PRODUCT PRODUCT INNER JOIN 
                    /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/MFGORDER MFGORDER INNER JOIN 
                        /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/OPERATION OPERATION INNER JOIN 
                            /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/CONTAINER CONTAINER INNER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/HISTORYCROSSREF HISTORYCROSSREF
                            ON HISTORYCROSSREF.TRACKINGID = CONTAINER.CONTAINERID INNER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/HISTORYMAINLINE HISTORYMAINLINE
                            ON HISTORYMAINLINE.HISTORYID = HISTORYCROSSREF.HISTORYID AND 
                                (HISTORYMAINLINE.TXNID >= HISTORYCROSSREF.STARTTXNID AND 
                                HISTORYMAINLINE.TXNID <= HISTORYCROSSREF.ENDTXNID) LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBTESTPLANHISTORY ZBTESTPLANHISTORY
                            ON HISTORYMAINLINE.HISTORYMAINLINEID = ZBTESTPLANHISTORY.HISTORYMAINLINEID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBTESTHISTORY TestHistory
                            ON ZBTESTPLANHISTORY.ZBTESTPLANHISTORYID = TestHistory.ZBTESTPLANHISTORYID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBTESTRESULTHISTORY ZBTESTRESULTHISTORY
                            ON TestHistory.ZBTESTHISTORYID = ZBTESTRESULTHISTORY.ZBTESTHISTORYID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBJUDGEMENTTYPE TestResultJudgement
                            ON ZBTESTRESULTHISTORY.ZBJUDGEMENTTYPEID = TestResultJudgement.ZBJUDGEMENTTYPEID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBJUDGEMENTTYPE TestJudgement
                            ON TestHistory.ZBJUDGEMENTTYPEID = TestJudgement.ZBJUDGEMENTTYPEID LEFT OUTER JOIN 
                                /shared/MES7/MES7PRD/MES7PRD_LMT_TMP_RO/ZBJUDGEMENTTYPE TestPlanJudgement
                            ON ZBTESTPLANHISTORY.ZBJUDGEMENTTYPEID = TestPlanJudgement.ZBJUDGEMENTTYPEID
                        ON OPERATION.OPERATIONID = HISTORYMAINLINE.OPERATIONID
                    ON CONTAINER.MFGORDERID = MFGORDER.MFGORDERID
                ON PRODUCT.PRODUCTID = MFGORDER.PRODUCTID
            ON PRODUCT.PRODUCTBASEID = PRODUCTBASE.PRODUCTBASEID
        ON CONTAINER.LEVELID = CONTAINERLEVEL.CONTAINERLEVELID
    ON HISTORYMAINLINE.RESOURCEID = RESOURCEDEF.RESOURCEID
WHERE
    (ZBTESTRESULTHISTORY.ZBTESTRESULTROWINDEX IN (0, 11) AND 
    OPERATION.OPERATIONID IN ('00060e2000006a64', '00060e2000006ac0')) AND 
    CONTAINERLEVEL.CONTAINERLEVELID = '00052820000000b2'
