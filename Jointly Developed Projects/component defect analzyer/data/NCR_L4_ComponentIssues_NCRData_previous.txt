PROCEDURE NCR_L4_ComponentIssues_NCRData_previous(
    IN p_StartDate DATE, 
    IN p_EndDate DATE, 
    IN p_SupplierName VARCHAR(40000), 
    IN p_ProductId VARCHAR(40000), 
    IN p_MfgOrderId VARCHAR(40000), 
    IN p_ComponentProductId VARCHAR(40000), 
    IN p_ComponentProductGroupId VARCHAR(40000),
    OUT result CURSOR (
        Product VARCHAR(255), 
        PartNumber VARCHAR(255), 
        ComponentProductGroup VARCHAR(255), 
        Component VARCHAR(255), 
        ComponentDesc VARCHAR(255), 
        MfgOrder VARCHAR(255), 
        ModuleSN VARCHAR(255), 
        TxnName VARCHAR(255), 
        ComponentIssueTxnDate TIMESTAMP, 
        QtyIssued INTEGER, 
        IssuedComponentReferenceDesignator VARCHAR(255), 
        Batch VARCHAR(255), 
        SupplierPartNumber VARCHAR(255), 
        SupplierLot VARCHAR(255), 
       NonConformanceReport VARCHAR(255), 
        NCRCreateDate TIMESTAMP, 
        NonConformanceReferenceDesignator VARCHAR(255), 
        FailModeReferenceDesignator VARCHAR(255), 
        DispostionCode VARCHAR(255), 
        RootCauseCode VARCHAR(255), 
        RootCauseCodeDesc VARCHAR(255), 
        RootCauseType VARCHAR(255), 
        NonConformanceCode VARCHAR(255), 
        NonConformanceCodeDesc VARCHAR(255), 
        NCRComment VARCHAR(255), 
        FailureModeCode VARCHAR(255), 
        FailureModeDesc VARCHAR(255), 
        NCRStatus INTEGER, 
        TxnMonth INTEGER, 
        TxnYear INTEGER, 
        MfgOrderID VARCHAR(255), 
        ProductID VARCHAR(255), 
        ComponentProductGroupID VARCHAR(255), 
        Supplier VARCHAR(255), 
        ComponentProductID VARCHAR(255)
        )
    )
    BEGIN
        OPEN result FOR 
            SELECT 
                TBL_Union.Product, 
				TBL_Union.PartNumber, 
				TBL_Union.ComponentProductGroup, 
				TBL_Union.Component, 
				TBL_Union.ComponentDesc, 
				TBL_Union.MfgOrder, 
				TBL_Union.ModuleSN, 
				TBL_Union.TxnName, 
				TBL_Union.ComponentIssueTxnDate, 
				TBL_Union.QtyIssued, 
				TBL_Union.IssuedComponentReferenceDesignator, 
				TBL_Union.Batch, 
				TBL_Union.SupplierPartNumber, 
				TBL_Union.SupplierLot, 
				max(TBL_Union.NonConformanceReport) as NonConformanceReport, 
				max(TBL_Union.NCRCreateDate) as NCRCreateDate, 
				max(TBL_Union.NonConformanceReferenceDesignator) as NonConformanceReferenceDesignator, 
				max(TBL_Union.FailModeReferenceDesignator) as FailModeReferenceDesignator, 
				max(TBL_Union.DispostionCode) as DispostionCode, 
				max(TBL_Union.RootCauseCode) as RootCauseCode, 
				max(TBL_Union.RootCauseCodeDesc) as RootCauseCodeDesc, 
				max(TBL_Union.RootCauseType) as RootCauseType, 
				max(TBL_Union.NonConformanceCode) as NonConformanceCode, 
				max(TBL_Union.NonConformanceCodeDesc) as NonConformanceCodeDesc, 
				max(TBL_Union.NCRComment) as NCRComment, 
				max(TBL_Union.FailureModeCode) as FailureModeCode, 
				max( TBL_Union.FailureModeDesc) as FailureModeDesc, 
				max(TBL_Union.NCRStatus) as NCRStatus, 
				TBL_Union.TxnMonth, 
				TBL_Union.TxnYear, 
				TBL_Union.MfgOrderID, 
				TBL_Union.ProductID, 
				TBL_Union.ComponentProductGroupID, 
				TBL_Union.Supplier, 
				TBL_Union.ComponentProductID
            FROM      
                /shared/MES7/"Component Defect Analysis"/ParametrizedQueriesMultipleParams/NCR_L3_UnionRwkAndNotRWKANDExcluding(p_StartDate, p_EndDate, p_SupplierName, p_ProductId, p_MfgOrderId, p_ComponentProductId, p_ComponentProductGroupId) TBL_Union 
 			GROUP BY
				TBL_Union.Product, 
				TBL_Union.PartNumber, 
				TBL_Union.ComponentProductGroup, 
				TBL_Union.Component, 
				TBL_Union.ComponentDesc, 
				TBL_Union.MfgOrder, 
				TBL_Union.ModuleSN, 
				TBL_Union.TxnName, 
				TBL_Union.ComponentIssueTxnDate, 
				TBL_Union.QtyIssued, 
				TBL_Union.IssuedComponentReferenceDesignator, 
				TBL_Union.Batch, 
				TBL_Union.SupplierPartNumber, 
				TBL_Union.SupplierLot,
				TBL_Union.TxnMonth, 
				TBL_Union.TxnYear, 
				TBL_Union.MfgOrderID, 
				TBL_Union.ProductID, 
				TBL_Union.ComponentProductGroupID, 
				TBL_Union.Supplier, 
				TBL_Union.ComponentProductID;
    END
